```typescript
function validatorUserName(rule: any, value: any, callback: any) {
  // rule: 校验规则对象
  // value：表单元素文本内容
  // callback：回调函数，如果符合条件就通过callback放行。如果不符合条件，callback注入错误信息
  if(value.length >= 3){
    callback();
  } else {
    callback(new Error('账号长度至少3位'))
  }
}

const rules ={
  username:[
    { required: true, trigger: 'change', validator: validatorUserName }
  ]
```
